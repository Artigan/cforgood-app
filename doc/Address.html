def open?
  todays = self.timetables.where(day: Time.now.wday)
  todays.each do |today|
    if (today.start_at.to_s(:time)..today.end_at.to_s(:time)).cover?(Time.now.to_s(:time))
      return true
    end
  end
  false
end
